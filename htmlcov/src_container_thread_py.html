32 def _connect_to_container(self): 

33 nwsettings = self.container.attrs['NetworkSettings'] 

34 self.container_gateway = nwsettings['Networks']['bridge']['Gateway'] 

35 self.container_ip = nwsettings['Networks']['bridge']['IPAddress'] 

36 logger.debug(self.container_ip) 

37 

38 ports = nwsettings['Ports'] 

39 assert len(ports) == 1 

40 

41 for port in ports.keys(): 

42 self.container_port = int(port.split('/')[0]) 

43 self.container_protocol = port.split('/')[1] 

44 logger.debug(self.container_port) 

45 logger.debug(self.container_protocol)

54 logger.debug(self.dest)
